146  /* `IsFor()` -> `IsNull()` の順で呼び出すことで2%高速化 */
147  if (itr->IsFor(board_key_) && !itr->IsNull()) {
148  if (itr->LookUp(hand_, depth_, len16, pn, dn, does_have_old_child)) {
149  amount = std::max(amount, itr->Amount());

154  } else if (itr->IsFor(board_key_, hand_)) {
155  if (itr->IsPossibleRepetition()) {

157  return SearchResult::MakeRepetition(hand_, len, amount, *depth_opt);
158  }
159  }
160

163  }
164  }
165  }

83 inline Hand ApplyDeltaHand(Hand target, Hand diff_src, Hand diff_dst) {
84  // 単純に実装するなら target + (diff_dst - diff_src) だが、オーバーフローの可能性があるので駒種別に判定する
85  Hand res = HAND_ZERO;

87  const auto target_cnt = hand_count(target, pr);
88  const auto src_cnt = hand_count(diff_src, pr);
89  const auto dst_cnt = hand_count(diff_dst, pr);
90  if (src_cnt < dst_cnt) {
91  const auto result_cnt = std::min(target_cnt + dst_cnt - src_cnt, PIECE_BIT_MASK[pr]);
92  add_hand(res, pr, result_cnt);
93  } else if (src_cnt > dst_cnt) {

95  add_hand(res, pr, result_cnt);

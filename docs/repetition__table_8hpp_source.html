201  void CollectGarbage() {

203  const auto erased_generation = generation_ - kGcKeepGeneration;
204 
205  // 対象entryのgenerationが [erased_generation, generation_] の範囲内ならfalse、それ以外ならtrue
206  const auto should_erase = [erased_generation, this](const TableEntry& entry) {

208 
209  // erased_generationとerased_generationの間にstd::uint32_t::maxの境界をまたいでいる可能性があるので注意
210  if (erased_generation < generation_) {
211  return entry_generation < erased_generation || generation_ < entry_generation;
212  } else {
213  return generation_ < entry_generation && entry_generation < erased_generation;
214  }
215  };

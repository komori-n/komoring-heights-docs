87 inline std::pair<PnDn, PnDn> InitialPnDnPlusOrNode(const Position& n, Move move) {
88  PnDn pn = tl_dfpn_plus_parameters.or_pn_base;
89  PnDn dn = tl_dfpn_plus_parameters.or_dn_base;
90 
91  const Color us = n.side_to_move();
92  const Color them = ~us;
93  const Square to = to_sq(move);
94  // n.attackers_to(color, to) はそこそこ重い処理なので、n.attackers_to(to) の結果を使い回すことで高速化できる
95  const auto attacker_bb = n.attackers_to(to);
96  const auto attack_support = (attacker_bb & n.pieces(us)).pop_count();
97  const auto defense_support = (attacker_bb & n.pieces(them)).pop_count();
98 
99  if (defense_support >= 2) {
100  // たくさん受け駒が利いている場合は後回し
101  pn += tl_dfpn_plus_parameters.or_defense;
102  }
103 
104  if (attack_support + (is_drop(move) ? 1 : 0) > defense_support) {
105  // 攻め駒がたくさんあるときは探索を優先する
106  dn += tl_dfpn_plus_parameters.or_support;
107  } else if (auto captured_pc = n.piece_on(to); captured_pc != NO_PIECE) {
108  const auto captured_pr = raw_type_of(captured_pc);
109  if (captured_pr == GOLD || captured_pr == SILVER) {
110  dn += tl_dfpn_plus_parameters.or_capture_gold_silver;
111  } else {
112  pn += tl_dfpn_plus_parameters.or_capture_others;
113  }
114  } else {
115  pn += tl_dfpn_plus_parameters.or_others;
116  }
117 
118  return {pn, dn};
119 }

127 inline std::pair<PnDn, PnDn> InitialPnDnPlusAndNode(const Position& n, Move move) {
128  const Color us = n.side_to_move();
129  const Color them = ~us;
130  const Square to = to_sq(move);
131  const Square king_sq = n.king_square(us);
132 
133  if (n.piece_on(to) != NO_PIECE) {
134  // コマを取る手は探索を優先する
135  return {tl_dfpn_plus_parameters.and_capture_pn, tl_dfpn_plus_parameters.and_capture_dn};
136  }
137 
138  if (!is_drop(move) && from_sq(move) == king_sq) {
139  // 玉を動かす手はそこそこ価値が高い
140  return {tl_dfpn_plus_parameters.and_king_pn, tl_dfpn_plus_parameters.and_king_dn};
141  }
142 
143  const auto attacker_bb = n.attackers_to(to);
144  const auto attack_support = (attacker_bb & n.pieces(them)).pop_count();
145  const auto defense_support = (attacker_bb & n.pieces(us)).pop_count();
146 
147  if (attack_support < defense_support + (is_drop(move) ? 1 : 0)) {
148  return {tl_dfpn_plus_parameters.and_good_pn, tl_dfpn_plus_parameters.and_good_dn};
149  }
150  return {tl_dfpn_plus_parameters.and_bad_pn, tl_dfpn_plus_parameters.and_bad_dn};
151 }

194 inline int MoveBriefEvaluation(const Node& n, Move move) {
195  auto us = n.Us();
196  auto king_sq = n.KingSquare();
197  auto to = to_sq(move);
198 
199  int value = 0;
200 
201  // 成れるのに成らない
202  if (!is_drop(move) && !is_promote(move)) {
203  auto from = from_sq(move);
204  auto before_pt = type_of(n.Pos().moved_piece_before(move));
205  if ((before_pt == PAWN || before_pt == BISHOP || before_pt == ROOK) &&
206  (enemy_field(us).test(from) || enemy_field(us).test(to))) {
207  value += 1000; // 歩、角、飛車を成らないのは大きく減点する（打ち歩詰めの時以外は考える必要ない）
208  }
209  }
210 
211  auto after_pt = type_of(n.Pos().moved_piece_after(move));
212  value -= detail::tl_pt_values[after_pt];
213  value += 10 * dist(king_sq, to);
214 
215  return value;
216 }

227 inline bool IsSumDeltaNode(const Node& n, Move move) {
228  const bool or_node = n.IsOrNode();
229  if (!is_drop(move)) {
230  // 駒打ち以外
231  if (or_node) {
232  const auto from = from_sq(move);
233  const auto to = to_sq(move);
234  const auto pc = n.Pos().piece_on(from);
235  const auto pt = type_of(pc);
236 
237  // 2 or 3段目の香成と不成
238  if (pt == LANCE) {
239  const auto king_sq = n.Pos().king_square(n.AndColor());
240  if ((n.Us() == BLACK && (rank_of(to) == RANK_3 || rank_of(to) == RANK_2) && king_sq == to + SQ_U) ||
241  (n.Us() == WHITE && (rank_of(to) == RANK_7 || rank_of(to) == RANK_8) && king_sq == to + SQ_D)) {
242  return false;
243  }
244  }
245  }
246  }
247 
248  return true;

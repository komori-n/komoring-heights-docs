77 void WaitSearchEnd() {
78  Timer timer;
79  timer.reset();
80 
81  const bool is_mate_search = Search::Limits.mate != 0;
82  const auto is_end = [&]() {
83  return Threads.stop || g_search_end || (is_mate_search && timer.elapsed() >= Search::Limits.mate);
84  };
85  constexpr TimePoint kTimePointMax = std::numeric_limits<TimePoint>::max();
86  const TimePoint pv_interval =
87  g_option.pv_interval > kTimePointMax ? kTimePointMax : static_cast<TimePoint>(g_option.pv_interval);
88 
89  TimePoint next_pv_out = pv_interval;
90  std::unique_lock<std::mutex> lock(g_end_mtx);
91  while (!is_end()) {
92  auto sleep_duration = 100;
93  if (next_pv_out < timer.elapsed() + sleep_duration) {
94  // このまま sleep_duration だけ寝ると予定時刻を過ぎてしまう
95  if (next_pv_out > timer.elapsed()) {
96  sleep_duration = next_pv_out - timer.elapsed();
97  } else {
98  sleep_duration = 1;
99  }
100  }
101 
102  g_end_cv.wait_for(lock, std::chrono::milliseconds(sleep_duration), is_end);
103  if (pv_interval > 0 && timer.elapsed() >= next_pv_out) {
104  g_searcher.RequestPrint();
105  next_pv_out = timer.elapsed() + pv_interval;
106  }
107  }
108 }

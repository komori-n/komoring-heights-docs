185 std::pair<NodeState, MateLen> KomoringHeights::SearchMainLoop(Node& n) {
186  NodeState node_state = NodeState::kUnknown;
187  auto len{kDepthMaxMateLen};
188 
189  for (Depth i = 0; i < kDepthMax; ++i) {
190  const auto result = SearchEntry(n, len);
191  const auto old_score = score_;
192  const auto score = Score::Make(option_.score_method, result, n.IsRootOrNode());
193 
194  if (monitor_.ShouldStop()) {
195  break;
196  }
197 
198  if (tl_thread_id == 0) {
199  score_ = score;
200  }
201  auto info = CurrentInfo();
202 
203  if (result.Pn() == 0) {
204  node_state = NodeState::kProven;
205 
206  // len 以下の手数の詰みが帰ってくるはず
207  KOMORI_PRECONDITION(result.Len().Len() <= len.Len());
208  if (tl_thread_id == 0) {
209  best_moves_ = pv_list_.BestMoves();
210  if (!option_.silent) {
211  sync_cout << CurrentInfo() << "# " << OrdinalNumber(i + 1) << " result: mate in " << best_moves_.size()
212  << "(upper_bound:" << result.Len() << ")" << sync_endl;
213  Print(n);
214  }
215 
216  if (option_.post_search_level == PostSearchLevel::kNone ||
217  (option_.post_search_level == PostSearchLevel::kUpperBound && best_moves_.size() == result.Len().Len())) {
218  break;
219  }
220  }
221 
222  if (result.Len().Len() <= 1) {
223  break;
224  }
225 
226  // 余詰探索に突入する
227  len = result.Len() - 2;
228  } else {
229  if (tl_thread_id == 0 && result.Dn() == 0 && result.Len() < len) {
230  sync_cout << info << "Failed to detect PV" << sync_endl;
231  }
232 
233  if (len != kDepthMaxMateLen) {
234  len = len + 2;
235  if (tl_thread_id == 0) {
236  best_moves_ = GetMatePath(n, len, true);
237  const auto final_result = SearchResult::MakeFinal<true>(n.OrHand(), len - 1, result.Amount());
238  pv_list_.Update(best_moves_[0], final_result, 0, best_moves_);
239  score_ = old_score;
240  }
241  } else {
242  node_state = result.GetNodeState();
243  }
244 
245  if (tl_thread_id == 0 && !option_.silent) {
246  sync_cout << info << "# " << OrdinalNumber(i + 1) << " result: " << result << sync_endl;
247  Print(n);
248  }
249  break;
250  }
251  }
252 
253  return {node_state, len};
254 }

256 SearchResult KomoringHeights::SearchEntry(Node& n, MateLen len) {
257  SearchResult result{};
258  PnDn thpn = (len == kDepthMaxMateLen) ? tl_thread_id : kInfinitePnDn;
259  PnDn thdn = (len == kDepthMaxMateLen) ? tl_thread_id : kInfinitePnDn;
260 
261  expansion_list_[tl_thread_id].Emplace(tt_, n, len, true, BitSet64::Full(), option_.multi_pv);
262  if (tl_thread_id == 0 && n.GetDepth() == 0) {
263  for (const auto& [move, result] : expansion_list_[0].Root().GetAllResults()) {
264  if (!result.IsFinal()) {
265  continue;
266  }
267 
268  // Final な手を見つけたとき、pv_list_ へその手順を記録しておく
269  UpdateFinalPv(n, move, result);
270  }
271  }
272  while (!monitor_.ShouldStop() && thpn <= kInfinitePnDn && thdn <= kInfinitePnDn) {
273  if (n.GetDepth() == 0) {
274  result = SearchImplForRoot(n, thpn, thdn, len);
275  } else {
276  std::uint32_t inc_flag = 0;
277  result = SearchImpl(n, thpn, thdn, len, inc_flag);
278  }
279  if (result.IsFinal() || monitor_.ShouldStop()) {
280  break;
281  }
282 
283  if (tl_thread_id == 0 && (result.Pn() >= kInfinitePnDn || result.Dn() >= kInfinitePnDn)) {
284  auto info = CurrentInfo();
285  sync_cout << info << "error: " << (result.Pn() >= kInfinitePnDn ? "pn" : "dn") << " overflow detected"
286  << sync_endl;
287  break;
288  }
289 
290  std::tie(thpn, thdn) = NextPnDnThresholds(result.Pn(), result.Dn(), thpn, thdn);
291  }
292 
293  auto query = tt_.BuildQuery(n);
294  query.SetResult(result);
295  expansion_list_[tl_thread_id].Pop();
296 
297  return result;
298 }

300 SearchResult KomoringHeights::SearchImplForRoot(Node& n, PnDn thpn, PnDn thdn, MateLen len) {
301  // 実装内容は SearchImpl() とほぼ同様なので詳しいロジックについてはそちらも参照。
302 
303  const auto orig_thpn = thpn;
304  const auto orig_thdn = thdn;
305  std::uint32_t inc_flag = 0;
306  auto& local_expansion = expansion_list_[tl_thread_id].Current();
307 
308  if (tl_thread_id == 0 && monitor_.ShouldPrint()) {
309  Print(n);
310  }
311  expansion_list_[tl_thread_id].EliminateDoubleCount(tt_, n);
312 
313  auto curr_result = local_expansion.CurrentResult(n);
314  if (local_expansion.DoesHaveOldChild()) {
315  inc_flag++;
316  ExtendSearchThreshold(curr_result, thpn, thdn);
317  }
318 
319  while (!monitor_.ShouldStop() && (curr_result.Pn() < thpn && curr_result.Dn() < thdn)) {
320  const auto best_move = local_expansion.BestMove();
321  const bool is_first_search = local_expansion.FrontIsFirstVisit();
322  const BitSet64 sum_mask = local_expansion.FrontSumMask();
323  const auto [child_thpn, child_thdn] = local_expansion.FrontPnDnThresholds(thpn, thdn);
324 
325  n.DoMove(best_move);
326  auto& child_expansion = expansion_list_[tl_thread_id].Emplace(tt_, n, len - 1, is_first_search, sum_mask);
327 
328  SearchResult child_result;
329  if (is_first_search) {
330  child_result = child_expansion.CurrentResult(n);
331  if (inc_flag > 0) {
332  inc_flag--;
333  }
334 
335  if (child_result.Pn() >= child_thpn || child_result.Dn() >= child_thdn) {
336  goto CHILD_SEARCH_END;
337  }
338  }
339  child_result = SearchImpl(n, child_thpn, child_thdn, len - 1, inc_flag);
340 
341  CHILD_SEARCH_END:
342  expansion_list_[tl_thread_id].Pop();
343  n.UndoMove();
344 
345  local_expansion.UpdateBestChild(child_result);
346  curr_result = local_expansion.CurrentResult(n);
347 
348  if (tl_thread_id == 0 && n.GetDepth() == 0 && child_result.IsFinal()) {
349  // Final な手を見つけたとき、pv_list_ へその手順を記録しておく
350  UpdateFinalPv(n, best_move, child_result);
351  }
352 
353  thpn = orig_thpn;
354  thdn = orig_thdn;
355 
356  if (inc_flag > 0) {
357  ExtendSearchThreshold(curr_result, thpn, thdn);
358  }
359  }
360 
361  return curr_result;
362 }

364 SearchResult KomoringHeights::SearchImpl(Node& n, PnDn thpn, PnDn thdn, MateLen len, std::uint32_t& inc_flag) {
365  const PnDn orig_thpn = thpn;
366  const PnDn orig_thdn = thdn;
367  const std::uint32_t orig_inc_flag = inc_flag;
368 
369  auto& local_expansion = expansion_list_[tl_thread_id].Current();
370  monitor_.Visit(n.GetDepth());
371  if (tl_thread_id == 0 && monitor_.ShouldPrint()) {
372  Print(n);
373  }
374 
375  if (n.GetDepth() >= kDepthMax) {
376  return SearchResult::MakeRepetition(n.OrHand(), len, 1, 0);
377  }
378 
379  expansion_list_[tl_thread_id].EliminateDoubleCount(tt_, n);
380 
381  // 必要があれば TCA による探索延長をしたいので、このタイミングで現局面の pn/dn を取得する。
382  auto curr_result = local_expansion.CurrentResult(n);
383  // Threshold Controlling Algorithm(TCA).
384  // 浅い結果を参照している場合、無限ループになる可能性があるので少しだけ探索を延長する
385  if (local_expansion.DoesHaveOldChild()) {
386  inc_flag++;
387  }
388 
389  if (inc_flag > 0) {
390  ExtendSearchThreshold(curr_result, thpn, thdn);
391  }
392 
393  if (tl_gc_thread && monitor_.ShouldCheckHashfull()) {
394  if (tt_.Hashfull() >= kExecuteGcHashfullThreshold) {
395  tt_.CollectGarbage(kGcRemovalRatio);
396  }
397  monitor_.ResetNextHashfullCheck();
398  }
399 
400  while (!monitor_.ShouldStop() && (curr_result.Pn() < thpn && curr_result.Dn() < thdn)) {
401  // local_expansion.BestMove() にしたがい子局面を展開する
402  // （curr_result.Pn() > 0 && curr_result.Dn() > 0 なので、BestMove が必ず存在する）
403  const auto best_move = local_expansion.BestMove();
404  const bool is_first_search = local_expansion.FrontIsFirstVisit();
405  const BitSet64 sum_mask = local_expansion.FrontSumMask();
406  const auto [child_thpn, child_thdn] = local_expansion.FrontPnDnThresholds(thpn, thdn);
407 
408  n.DoMove(best_move);
409 
410  // 子局面を展開する。展開した expansion は UndoMove() の直前に忘れずに開放しなければならない。
411  auto& child_expansion = expansion_list_[tl_thread_id].Emplace(tt_, n, len - 1, is_first_search, sum_mask);
412 
413  SearchResult child_result;
414  if (is_first_search) {
415  child_result = child_expansion.CurrentResult(n);
416  // 新規局面を展開したので、inc_flag を 1 つ減らしておく
417  // オリジナルの TCA では inc_flag は bool 値なので単に false を代入していたが、KomoringHeights では
418  // 非負整数へ拡張している。
419  if (inc_flag > 0) {
420  inc_flag--;
421  }
422 
423  // 子局面を初展開する場合、child_result を計算した時点で threshold を超過する可能性がある
424  // しかし、SearchImpl をコールしてしまうと TCA の探索延長によりすぐに返ってこない可能性がある
425  // ゆえに、この時点で Exceed している場合は SearchImpl を呼ばないようにする。
426  if (child_result.Pn() >= child_thpn || child_result.Dn() >= child_thdn) {
427  goto CHILD_SEARCH_END;
428  }
429  }
430  child_result = SearchImpl(n, child_thpn, child_thdn, len - 1, inc_flag);
431 
432  CHILD_SEARCH_END:
433  expansion_list_[tl_thread_id].Pop();
434  n.UndoMove();
435 
436  local_expansion.UpdateBestChild(child_result);
437  curr_result = local_expansion.CurrentResult(n);
438 
439  // TCA で延長したしきい値はいったん戻す
440  thpn = orig_thpn;
441  thdn = orig_thdn;
442  if (inc_flag > 0) {
443  // TCA 継続中ならしきい値を伸ばす
444  ExtendSearchThreshold(curr_result, thpn, thdn);
445  } else if (inc_flag == 0 && orig_inc_flag > 0) {
446  // TCA の展開が終わったので、いったん親局面に戻る
447  break;
448  }
449  }
450 
452  inc_flag = std::min(inc_flag, orig_inc_flag);
453  return curr_result;
454 }

456 std::vector<Move> KomoringHeights::GetMatePath(Node& n, MateLen len, bool exact) {
457  std::vector<Move> best_moves;
458  pv_search_ = true;
459  while (len.Len() > 0) {
460  // 1手詰はTTに書かれていない可能性があるので先にチェックする
461  const auto [move, hand] = CheckMate1Ply(n);
462  if (move != MOVE_NONE) {
463  best_moves.push_back(move);
464  n.DoMove(move);
465  break;
466  }
467 
468  // 子ノードの中から最善っぽい手を選ぶ
469  const auto [best_move, next_len] =
470  (n.IsOrNode() ? GetBestMoveOrNode(n, len, exact) : GetBestMoveAndNode(n, len, exact));
471 
472  if (best_move == MOVE_NONE) {
473  break;
474  }
475 
476  len = next_len;
477  n.DoMove(best_move);
478  best_moves.push_back(best_move);
479  }
480  pv_search_ = false;
481 
482  RollBack(n, best_moves);
483  return best_moves;
484 }

486 std::pair<Move, MateLen> KomoringHeights::GetBestMoveOrNode(Node& n, MateLen len, bool exact) {
487  KOMORI_PRECONDITION(n.IsOrNode());
488  if (!exact) {
489  const auto [move, proven_len] = LookUpBestMoveOrNode(tt_, n);
490  if (proven_len + 1 <= len) { // proven_len <= len - 1
491  return {move, proven_len};
492  }
493  }
494 
495  auto& expansion = expansion_list_[tl_thread_id].Emplace(tt_, n, len, true, BitSet64::Full(), option_.multi_pv);
496  std::uint32_t inc_flag = 0;
497  SearchImpl(n, kInfinitePnDn, kInfinitePnDn, len, inc_flag);
498  // exclude を無視して最善手を取りたいので、expansion.BestMove() は使えないので注意。
499  const auto [move, result] = *expansion.GetAllResults().begin();
500  expansion_list_[tl_thread_id].Pop();
501 
502  // exact, !exact の両方のケースで len 手以下の詰みになるような手を返せば良い
503  if (result.Pn() != 0 || result.Len() + 1 > len) {
504  std::this_thread::sleep_for(std::chrono::seconds(1));
505  return {MOVE_NONE, kDepthMaxMateLen};
506  }
507 
508  return {move, result.Len()};

511 std::pair<Move, MateLen> KomoringHeights::GetBestMoveAndNode(Node& n, MateLen len, bool exact) {
512  KOMORI_PRECONDITION(!n.IsOrNode());
513  if (exact) {
514  auto& expansion = expansion_list_[tl_thread_id].Emplace(tt_, n, len - 2, true, BitSet64::Full(), option_.multi_pv);
515  std::uint32_t inc_flag = 0;
516  SearchImpl(n, kInfinitePnDn, kInfinitePnDn, len - 2, inc_flag);
517  // exclude を無視して最善手を取りたいので、expansion.BestMove() は使えないので注意。
518  const auto [move2, result] = *expansion.GetAllResults().begin();
519  expansion_list_[tl_thread_id].Pop();
520 
521  // len - 2 手の探索で詰みを逃れる手が見つかるはずなので、それを返す
522  if (result.Dn() != 0) {
523  sync_cout << "info string error: failed to get best move (and node)@" << n.GetDepth() << sync_endl;
524  return {MOVE_NONE, kDepthMaxMateLen};
525  }
526  return {move2, len - 1};
527  } else {
528  const auto [move, proven_len] = LookUpBestMoveAndNode(tt_, n);
529  if (proven_len + 1 <= len) { // proven_len <= len - 1
530  return {move, proven_len};
531  }
532 
533  auto& expansion = expansion_list_[tl_thread_id].Emplace(tt_, n, len, true, BitSet64::Full(), option_.multi_pv);
534  std::uint32_t inc_flag = 0;
535  SearchImpl(n, kInfinitePnDn, kInfinitePnDn, len, inc_flag);
536  // exclude を無視して最善手を取りたいので、expansion.BestMove() は使えないので注意。
537  const auto [move2, result] = *expansion.GetAllResults().begin();
538  expansion_list_[tl_thread_id].Pop();
539 
540  if (result.Pn() != 0 || result.Len() + 1 > len) {
541  sync_cout << "info string error: failed to get best move (and node)@" << n.GetDepth() << sync_endl;
542  return {MOVE_NONE, kDepthMaxMateLen};

548 void KomoringHeights::UpdateFinalPv(Node& n, Move move, const SearchResult& result) {

570  // 余詰探索完了後に GetMatePath(n, len, true) を呼び出すことで最終的な詰み手順を構成する

590  const auto& root = expansion_list_[0].Root();

597  // pv_list_ を昇順に並び替えるために、multi_pv_ の値に関係なくすべての子の探索結果を更新しなければならない
598  for (const auto& [move, result] : root.GetAllResults()) {
599  if (move == root.BestMove() || result.IsFinal()) {
600  // best_move -> 上で更新済み
601  // final -> SearchImplForRoot で更新済み
602  continue;
603  }
604 
605  pv_list_.Update(move, result);
606  }
607  } else if (!pv_list_.BestMoves().empty()) {
608  usi_output.Set(UsiInfoKey::kCurrMove, USI::move(pv_list_.BestMoves()[0]));
609  }
610 
611  for (const auto& pv_info : Take(pv_list_.GetPvList(), option_.multi_pv)) {
612  auto score = Score::Make(option_.score_method, pv_info.result, n.IsRootOrNode());
613  score.AddOneIfFinal();
614  usi_output.PushPVBack(pv_info.depth, score.ToString(), ToString(pv_info.pv));
615  }
616

618 }

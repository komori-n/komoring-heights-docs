23 std::pair<Move, MateLen> LookUpBestMove(tt::TranspositionTable& tt, Node& n, MateLen len) {
24  Move best_move = MOVE_NONE;
25  MateLen best_len = n.IsOrNode() ? kDepthMaxMateLen : kZeroMateLen;
26  MateLen best_disproven_len = kZeroMateLen;
27  for (const auto move : MovePicker{n}) {
28  const auto query = tt.BuildChildQuery(n, move.move);
29  const auto [disproven_len, proven_len] = query.FinalRange();
30  if (n.IsOrNode() && proven_len < best_len) {
31  best_move = move.move;
32  best_len = proven_len;
33  best_disproven_len = disproven_len;
34  } else if (!n.IsOrNode()) {
35  if (proven_len > best_len || (proven_len == best_len && best_disproven_len < disproven_len)) {
36  best_move = move.move;
37  best_len = proven_len;
38  best_disproven_len = disproven_len;
39  }
40  }
41  }
42 
43  if (len - 1 < best_len) {
44  best_move = MOVE_NONE;
45  }
46 
47  return {best_move, best_len};
48 }

163 std::pair<NodeState, MateLen> KomoringHeights::SearchMainLoop(Node& n) {
164  auto node_state{NodeState::kUnknown};
165  auto len{kDepthMaxMateLen};
166 
167  for (Depth i = 0; i < kDepthMax; ++i) {
168  // `result` が余詰探索による不詰だったとき、後から元の状態（詰み）に戻せるようにしておく
169  const auto old_score = score_;
170  const auto result = SearchEntry(n, len);
171  score_ = Score::Make(option_.score_method, result, n.IsRootOrNode());
172 
173  auto info = CurrentInfo();
174 
175  if (result.Pn() == 0) {
176  // len 以下の手数の詰みが帰ってくるはず
177  KOMORI_PRECONDITION(result.Len().Len() <= len.Len());
178  best_moves_ = GetMatePath(n, result.Len());
179 
180  sync_cout << info << "# " << OrdinalNumber(i + 1) << " result: mate in " << best_moves_.size()
181  << "(upper_bound:" << result.Len() << ")" << sync_endl;
182  std::ostringstream oss;
183  for (const auto move : best_moves_) {
184  oss << move << " ";
185  }
186  info.Set(UsiInfoKey::kPv, oss.str());
187  sync_cout << info << sync_endl;
188  node_state = NodeState::kProven;
189  if (result.Len().Len() <= 1) {
190  break;
191  }
192 
193  if (option_.post_search_level == PostSearchLevel::kNone ||
194  (option_.post_search_level == PostSearchLevel::kUpperBound && best_moves_.size() == result.Len().Len())) {
195  break;
196  }
197 
198  len = result.Len() - 2;
199  } else {
200  sync_cout << info << "# " << OrdinalNumber(i + 1) << " result: " << result << sync_endl;
201  if (result.Dn() == 0 && result.Len() < len) {
202  sync_cout << info << "Failed to detect PV" << sync_endl;
203  }
204  if (node_state == NodeState::kProven) {
205  len = len + 2;
206  score_ = old_score;
207  best_moves_ = GetMatePath(n, len);
208  }
209  break;
210  }
211  }
212 
213  return {node_state, len};
214 }

216 SearchResult KomoringHeights::SearchEntry(Node& n, MateLen len) {
217  SearchResult result{};
218  PnDn thpn = (len == kDepthMaxMateLen) ? 1 : kInfinitePnDn;
219  PnDn thdn = (len == kDepthMaxMateLen) ? 1 : kInfinitePnDn;
220 
221  expansion_list_.Emplace(tt_, n, len, true);
222  while (!monitor_.ShouldStop() && thpn <= kInfinitePnDn && thdn <= kInfinitePnDn) {
223  std::uint32_t inc_flag = 0;
224  result = SearchImpl(n, thpn, thdn, len, inc_flag);
225  if (result.IsFinal()) {
226  break;
227  }
228 
229  if (result.Pn() >= kInfinitePnDn || result.Dn() >= kInfinitePnDn) {
230  auto info = CurrentInfo();
231  sync_cout << info << "error: " << (result.Pn() >= kInfinitePnDn ? "pn" : "dn") << " overflow detected"
232  << sync_endl;
233  break;
234  }
235 
236  score_ = Score::Make(option_.score_method, result, n.IsRootOrNode());
237  // 反復深化のしきい値を適当に伸ばす
238  thpn = ClampPnDn(thpn, SaturatedMultiply<PnDn>(result.Pn(), 2), kInfinitePnDn);
239  thdn = ClampPnDn(thdn, SaturatedMultiply<PnDn>(result.Dn(), 2), kInfinitePnDn);
240  }
241  expansion_list_.Pop();
242 
243  auto query = tt_.BuildQuery(n);
244  query.SetResult(result);
245 
246  return result;
247 }

249 SearchResult KomoringHeights::SearchImpl(Node& n, PnDn thpn, PnDn thdn, MateLen len, std::uint32_t& inc_flag) {
250  const PnDn orig_thpn = thpn;
251  const PnDn orig_thdn = thdn;
252  const std::uint32_t orig_inc_flag = inc_flag;
253 
254  auto& local_expansion = expansion_list_.Current();
255  monitor_.Visit(n.GetDepth());
256  PrintIfNeeded(n);
257 
258  if (n.GetDepth() >= kDepthMax) {
259  return SearchResult::MakeRepetition(n.OrHand(), len, 1, 0);
260  }
261 
262  expansion_list_.EliminateDoubleCount(tt_, n);
263 
264  // 必要があれば TCA による探索延長をしたいので、このタイミングで現局面の pn/dn を取得する。
265  auto curr_result = local_expansion.CurrentResult(n);
266  // Threshold Controlling Algorithm(TCA).
267  // 浅い結果を参照している場合、無限ループになる可能性があるので少しだけ探索を延長する
268  if (local_expansion.DoesHaveOldChild()) {
269  inc_flag++;
270  }
271 
272  if (inc_flag > 0) {
273  ExtendSearchThreshold(curr_result, thpn, thdn);
274  }
275 
276  if (n.GetDepth() > 0 && monitor_.ShouldCheckHashfull()) {
277  if (tt_.Hashfull() >= kExecuteGcHashfullThreshold) {
278  tt_.CollectGarbage(kGcRemovalRatio);
279  }
280  monitor_.ResetNextHashfullCheck();
281  }
282 
283  while (!monitor_.ShouldStop() && (curr_result.Pn() < thpn && curr_result.Dn() < thdn)) {
284  // local_expansion.BestMove() にしたがい子局面を展開する
285  // （curr_result.Pn() > 0 && curr_result.Dn() > 0 なので、BestMove が必ず存在する）
286  const auto best_move = local_expansion.BestMove();
287  const bool is_first_search = local_expansion.FrontIsFirstVisit();
288  const BitSet64 sum_mask = local_expansion.FrontSumMask();
289  const auto [child_thpn, child_thdn] = local_expansion.FrontPnDnThresholds(thpn, thdn);
290 
291  n.DoMove(best_move);
292 
293  // 子局面を展開する。展開した expansion は UndoMove() の直前に忘れずに開放しなければならない。
294  auto& child_expansion = expansion_list_.Emplace(tt_, n, len - 1, is_first_search, sum_mask);
295 
296  SearchResult child_result;
297  if (is_first_search) {
298  child_result = child_expansion.CurrentResult(n);
299  // 新規局面を展開したので、inc_flag を 1 つ減らしておく
300  // オリジナルの TCA では inc_flag は bool 値なので単に false を代入していたが、KomoringHeights では
301  // 非負整数へ拡張している。
302  if (inc_flag > 0) {
303  inc_flag--;
304  }
305 
306  // 子局面を初展開する場合、child_result を計算した時点で threshold を超過する可能性がある
307  // しかし、SearchImpl をコールしてしまうと TCA の探索延長によりすぐに返ってこない可能性がある
308  // ゆえに、この時点で Exceed している場合は SearchImpl を呼ばないようにする。
309  if (child_result.Pn() >= child_thpn || child_result.Dn() >= child_thdn) {
310  goto CHILD_SEARCH_END;
311  }
312  }
313  child_result = SearchImpl(n, child_thpn, child_thdn, len - 1, inc_flag);
314 
315  CHILD_SEARCH_END:
316  expansion_list_.Pop();
317  n.UndoMove();
318 
319  local_expansion.UpdateBestChild(child_result);
320  curr_result = local_expansion.CurrentResult(n);
321 
322  // TCA で延長したしきい値はいったん戻す
323  thpn = orig_thpn;
324  thdn = orig_thdn;
325  if (inc_flag > 0) {
326  // TCA 継続中ならしきい値を伸ばす
327  ExtendSearchThreshold(curr_result, thpn, thdn);
328  } else if (inc_flag == 0 && orig_inc_flag > 0) {
329  // TCA の展開が終わったので、いったん親局面に戻る
330  break;
331  }
332  }
333 
335  inc_flag = std::min(inc_flag, orig_inc_flag);
336  return curr_result;
337 }

339 std::vector<Move> KomoringHeights::GetMatePath(Node& n, MateLen len) {
340  std::vector<Move> best_moves;
341  while (len.Len() > 0) {
342  // 1手詰はTTに書かれていない可能性があるので先にチェックする
343  const auto [move, hand] = CheckMate1Ply(n);
344  if (move != MOVE_NONE) {
345  best_moves.push_back(move);
346  n.DoMove(move);
347  break;
348  }
349 
350  // 子ノードの中から最善っぽい手を選ぶ
351  auto [best_move, new_len] = LookUpBestMove(tt_, n, len);
352  if (best_move == MOVE_NONE) {
353  tt_.NewSearch();
354  SearchEntry(n, len);
355 
356  std::tie(best_move, new_len) = LookUpBestMove(tt_, n, len);
357  if (best_move == MOVE_NONE) {
358  break;
359  }
360  }
361 
362  len = new_len;
363  n.DoMove(best_move);
364  best_moves.push_back(best_move);
365  }
366 
367  RollBack(n, best_moves);
368  return best_moves;
369 }

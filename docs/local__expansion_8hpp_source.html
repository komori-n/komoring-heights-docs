102  auto MakeComparer() const {
103  const SearchResultComparer sr_comparer{or_node_};
104  return [this, sr_comparer](std::size_t i_raw, std::size_t j_raw) -> bool {
105  // `SearchResultComparer` で大小比較の決着がつくならそれに従う。
106  // `SearchResultComparer` で結論がでなければ、指し手自体の評価値（指し手生成時に付与）で大小を決める。
107  const auto& left_result = results_[i_raw];
108  const auto& right_result = results_[j_raw];
109  const auto ordering = sr_comparer(left_result, right_result);
110  if (ordering == SearchResultComparer::Ordering::kLess) {
111  return true;
112  } else if (ordering == SearchResultComparer::Ordering::kGreater) {
113  return false;
114  }
115 
116  return mp_[i_raw].value < mp_[j_raw].value;
117  };
118  }

143  for (const auto& [i_raw, move] : WithIndex<std::uint32_t>(mp_)) {
144  const auto hand_after = n.OrHandAfter(move.move);

146  auto& result = results_[i_raw];
147  auto& query = queries_[i_raw];
148

162  result =

164

166  nn.DoMove(move.move);
167  if (auto res = detail::CheckObviousFinalOrNode(nn); res.has_value()) {
168  result = *res;
169  query.SetResult(*res);
170  }
171  nn.UndoMove();
172  }
173

176  sum_mask_.Reset(i_raw);
177  }
178

180  while (next_dep.has_value()) {

183  idx_.Pop();

188  }

411  constexpr PnDn GetDelta() const {
412  if (idx_.empty()) {
413  return 0;
414  }
415 
416  const auto& best_result = FrontResult();
417  // 差分計算用の値を予め持っているので、高速に計算できる
418  auto sum_delta = sum_delta_except_best_;
419  auto max_delta = max_delta_except_best_;
420  if (sum_mask_[idx_.front()]) {
421  sum_delta = ClampPnDn(sum_delta + best_result.Delta(or_node_));
422  } else {
423  max_delta = std::max(max_delta, best_result.Delta(or_node_));
424  }
425 
426  // 後回しにしている子局面が存在する場合、その値をδ値に加算しないと局面を過大評価してしまう。
427  //
428  // 例） sfen +P5l2/4+S4/p1p+bpp1kp/6pgP/3n1n3/P2NP4/3P1NP2/2P2S3/3K3L1 b RGSL2Prb2gsl3p 159
429  // 1筋の合駒を考える時、玉方が合駒を微妙に変えることで読みの深さを指数関数的に大きくできてしまう

433  sum_delta += std::max<std::size_t>((mp_.size() - idx_.size()) / 8, 1);

523  SearchResult GetDisprovenResult(const Node& n) const {
524  // children_ は千日手エントリが手前に来るようにソートされているので、以下のようにして千日手判定ができる
525  if (!mp_.empty()) {
526  if (const auto& result = FrontResult(); result.GetFinalData().IsRepetition()) {
527  const auto depth = result.GetFinalData().repetition_start;
528  const auto amount = result.Amount();
529  if (depth < n.GetDepth()) {
530  return SearchResult::MakeRepetition(n.OrHand(), len_, amount, depth);
531  } else {
532  return SearchResult::MakeFinal<false>(n.OrHand(), len_, amount);
533  }
534  }
535  }
536 
537  // フツーの不詰
538  if (or_node_) {
539  // 子局面の反証駒の極大集合を計算する
540  HandSet set{DisproofHandTag{}};
541  MateLen mate_len = kDepthMaxMateLen;
542  SearchAmount amount = 1;
543  for (const auto i_raw : idx_) {
544  const auto& result = results_[i_raw];
545  const auto child_move = mp_[i_raw];
546 
547  set.Update(BeforeHand(n.Pos(), child_move, result.GetFinalData().hand));
548  amount = std::max(amount, result.Amount());
549  if (result.Len() + 1 < mate_len) {
550  mate_len = result.Len() + 1;
551  }
552  }
553  amount += std::max<SearchAmount>(mp_.size(), 1) - 1;
554  const auto disproof_hand = set.Get(n.Pos());
555 
556  return SearchResult::MakeFinal<false>(disproof_hand, mate_len, amount);
557  } else {
558  const auto& result = FrontResult();
559  auto disproof_hand = result.GetFinalData().hand;
560  const auto mate_len = std::min(result.Len() + 1, kDepthMaxMateLen);
561  const auto amount = result.Amount() + mp_.size() - 1;
562 
563  // 駒打ちならその駒を持っていないといけない
564  if (const auto best_move = mp_[idx_[0]]; is_drop(best_move)) {
565  const auto pr = move_dropped_piece(best_move);
566  const auto pr_cnt = hand_count(MergeHand(n.OrHand(), n.AndHand()), pr);
567  const auto disproof_pr_cnt = hand_count(disproof_hand, pr);
568  if (pr_cnt - disproof_pr_cnt <= 0) {

571  sub_hand(disproof_hand, pr, disproof_pr_cnt);
572  add_hand(disproof_hand, pr, pr_cnt - 1);
573  }
574  }

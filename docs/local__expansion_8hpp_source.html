111  const SearchResultComparer sr_comparer{or_node_};
112  return [this, sr_comparer](std::size_t i_raw, std::size_t j_raw) -> bool {
113  // `SearchResultComparer` で大小比較の決着がつくならそれに従う。
114  // `SearchResultComparer` で結論がでなければ、指し手自体の評価値（指し手生成時に付与）で大小を決める。
115  const auto& left_result = results_[i_raw];
116  const auto& right_result = results_[j_raw];
117  const auto ordering = sr_comparer(left_result, right_result);
118  if (ordering == SearchResultComparer::Ordering::kLess) {
119  return true;
120  } else if (ordering == SearchResultComparer::Ordering::kGreater) {
121  return false;
122  }
123

125  };

466  PnDn GetDelta() const {
467  if (idx_.empty()) {
468  return 0;

471  const auto& best_result = FrontResult();
472  // 差分計算用の値を予め持っているので、高速に計算できる
473  auto sum_delta = sum_delta_except_best_;
474  auto max_delta = max_delta_except_best_;
475  if (sum_mask_[idx_[excluded_moves_]]) {
476  sum_delta = ClampPnDn(sum_delta + best_result.Delta(or_node_));
477  } else {

482  //
483  // 例） sfen +P5l2/4+S4/p1p+bpp1kp/6pgP/3n1n3/P2NP4/3P1NP2/2P2S3/3K3L1 b RGSL2Prb2gsl3p 159

548  SearchResult GetWinResult(const Node& n) const {
549  // excluded_moves_ に関係なく最も良い手がほしいので FrontResult() は使えない
550  const auto& result = results_[idx_.front()];
551  const auto best_move = mp_[idx_[0]];
552  const auto mate_len = result.Len() + 1;
553  const auto amount = result.Amount() + mp_.size() - 1;
554  const auto after_hand = result.GetFinalData().hand;
555 
556  if (or_node_) {
557  const auto proof_hand = BeforeHand(n.Pos(), best_move, after_hand);
558  return SearchResult::MakeFinal<true>(proof_hand, mate_len, amount);
559  } else {
560  auto disproof_hand = after_hand;
561 
562  // 駒打ちならその駒を持っていないといけない
563  if (is_drop(best_move)) {
564  const auto pr = move_dropped_piece(best_move);
565  const auto pr_cnt = hand_count(MergeHand(n.OrHand(), n.AndHand()), pr);
566  const auto disproof_pr_cnt = hand_count(disproof_hand, pr);
567  if (pr_cnt - disproof_pr_cnt <= 0) {
568  // もし現局面の攻め方の持ち駒が disproof_hand だった場合、打とうとしている駒 pr が攻め方に独占されているため
569  // 受け方は BestMove() を着手することができない。そのため、攻め方の持ち駒を1枚受け方に渡す必要がある。
570 
571  // 現局面で OR node + AND node の pr の持ち駒枚数を N とすると、OR node 側の pr の枚数が N-1 になるように
572  // disproof_hand を調整する。このとき、disproof_hand の pr 枚数が N より大きい可能性があるので注意。
573  sub_hand(disproof_hand, pr, disproof_pr_cnt - pr_cnt + 1);
574  }
575  }
576 
577  // 千日手のときは MakeRepetition で返す
578  if (result.GetFinalData().IsRepetition()) {
579  const auto depth = result.GetFinalData().repetition_start;
580  if (depth < n.GetDepth()) {
581  return SearchResult::MakeRepetition(n.OrHand(), mate_len, amount, depth);
582  }
583  }
584  return SearchResult::MakeFinal<false>(disproof_hand, mate_len, amount);
585  }

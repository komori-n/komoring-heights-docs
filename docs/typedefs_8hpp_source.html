140 constexpr inline T SaturatedMultiply(T lhs, T rhs) noexcept {
141  constexpr T kMax = std::numeric_limits<T>::max();
142  constexpr T kMin = std::numeric_limits<T>::min();
143 
144  if (lhs == 0 || rhs == 0) {
145  return 0;
146  } else if (lhs > 0 && rhs > 0) {
147  if (kMax / lhs < rhs) {
148  return kMax;
149  }
150  } else if (lhs < 0 && rhs < 0) {

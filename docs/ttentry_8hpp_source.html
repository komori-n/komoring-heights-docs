320  constexpr bool LookUp(Hand hand, Depth depth, MateLen16& len, PnDn& pn, PnDn& dn, bool& use_old_child)
321  const noexcept {
322  const auto depth16 = static_cast<std::int16_t>(depth);
323 
324  // 1. 現局面とエントリが一致
325  if (hand_ == hand) {
326  return LookUpExact(depth16, len, pn, dn, use_old_child);
327  }
328 
329  // 2. 現局面が劣等局面
330  const bool is_inferior = hand_is_equal_or_superior(hand_, hand);
331  if (is_inferior) {
332  // 劣等かつ優等な局面は一致局面だけ。つまり 3. の if 文は省略できる。
333  return LookUpInferior(depth16, len, pn, dn, use_old_child);
334  }
335 
336  // 3. 現局面が優等局面
337  const bool is_superior = hand_is_equal_or_superior(hand, hand_);
338  if (is_superior) {
339  return LookUpSuperior(depth16, len, pn, dn, use_old_child);
340  }
341 
342  // 優等でも劣等でもない局面。何もせずに返る
343  return false;

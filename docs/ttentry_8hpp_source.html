378  bool LookUp(Hand hand, Depth depth, MateLen len, PnDn& pn, PnDn& dn, bool& use_old_child) const noexcept {
379  const auto depth16 = static_cast<std::int16_t>(depth);
380 
381  // 1. 現局面とエントリが一致
382  const Hand entry_hand = hand_.load(std::memory_order_relaxed);
383  if (entry_hand == hand) {
384  return LookUpExact(depth16, len, pn, dn, use_old_child);
385  }
386 
387  // 2. 現局面が劣等局面
388  const bool is_inferior = hand_is_equal_or_superior(entry_hand, hand);
389  if (is_inferior) {
390  // 劣等かつ優等な局面は一致局面だけ。つまり 3. の if 文は省略できる。
391  return LookUpInferior(depth16, len, pn, dn, use_old_child);
392  }
393 
394  // 3. 現局面が優等局面
395  const bool is_superior = hand_is_equal_or_superior(hand, entry_hand);
396  if (is_superior) {
397  return LookUpSuperior(depth16, len, pn, dn, use_old_child);
398  }
399 
400  // 優等でも劣等でもない局面。何もせずに返る
401  return false;
402  }

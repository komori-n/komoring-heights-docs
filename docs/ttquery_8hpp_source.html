120  SearchResult LookUp(bool& does_have_old_child, MateLen len, InitialEvalFunc&& eval_func) const {
121  PnDn pn = 1;
122  PnDn dn = 1;
123  SearchAmount amount = 1;
124 
125  bool found_exact = false;
126  BitSet64 sum_mask = BitSet64::Full();
127 
128  for (auto itr = initial_entry_pointer_; !itr->IsNull(); ++itr) {
129  std::shared_lock lock(*itr);
130  // 本来は lock 後にも !itr->Null() のチェックが必要だが、itr->Hand() == kNullHand のとき itr->LookUp() が必ず
131  // 失敗するので、このタイミングでのチェックは省略できる。
132  if (itr->IsFor(board_key_)) {
133  if (itr->LookUp(hand_, depth_, len, pn, dn, does_have_old_child)) {
134  amount = std::max(amount, itr->Amount());
135  if (pn == 0) {
136  return SearchResult::MakeFinal<true>(itr->GetHand(), len, amount);
137  } else if (dn == 0) {
138  return SearchResult::MakeFinal<false>(itr->GetHand(), len, amount);
139  } else if (itr->IsFor(board_key_, hand_)) {
140  if (itr->IsPossibleRepetition()) {
141  if (auto opt = rep_table_->Contains(path_key_, len)) {
142  const auto [depth, table_len] = opt.value();
143  return SearchResult::MakeRepetition(hand_, table_len, amount, depth);
144  }
145  }
146 
147  found_exact = true;
148  sum_mask = itr->SumMask();
149  cached_entry_ = &*itr;
150  }
151  }
152  }
153  }
154 
155  // LookUp() 結果が final でないとき、スレッドごとに決められた周期で pn, dn へノイズを載せる
156  if (detail::tt_noise_timing != std::numeric_limits<std::uint32_t>::max()) {
157  if (detail::tt_noise_timing-- == 0) {
158  detail::tt_noise_timing = detail::tt_noise_interval;
159  // (pn, dn) へノイズを載せる
160  pn++;
161  dn++;
162  }
163  }
164 
165  if (found_exact) {
166  return SearchResult::MakeUnknown(pn, dn, len, amount, sum_mask);
167  }
168 
169  const auto [init_pn, init_dn] = std::forward<InitialEvalFunc>(eval_func)();
170  pn = std::max(pn, init_pn);
171  dn = std::max(dn, init_dn);
172 
173  return SearchResult::MakeFirstVisit(pn, dn, len, amount);
174  }

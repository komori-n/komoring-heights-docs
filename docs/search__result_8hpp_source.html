258  constexpr Ordering operator()(const SearchResult& lhs, const SearchResult& rhs) const noexcept {
259  // NOLINTBEGIN(bugprone-branch-clone)
260  if (lhs.Phi(or_node_) < rhs.Phi(or_node_)) {
261  return Ordering::kLess;
262  } else if (lhs.Phi(or_node_) > rhs.Phi(or_node_)) {
263  return Ordering::kGreater;
264  } else if (lhs.Delta(or_node_) < rhs.Delta(or_node_)) {
265  return Ordering::kLess;
266  } else if (lhs.Delta(or_node_) > rhs.Delta(or_node_)) {
267  return Ordering::kGreater;
268  }
269  // NOLINTEND(bugprone-branch-clone)
270 
271  if (lhs.Pn() == 0 /* && rhs.Pn() == 0 */) {
272  if (lhs.Len() < rhs.Len()) {
273  return or_node_ ? Ordering::kLess : Ordering::kGreater;
274  } else if (lhs.Len() > rhs.Len()) {
275  return or_node_ ? Ordering::kGreater : Ordering::kLess;
276  }
277  }
278 
279  if (lhs.Dn() == 0 /* && rhs.Dn() == 0 */) {
280  const auto l_rep_start = lhs.GetFinalData().repetition_start;
281  const auto r_rep_start = rhs.GetFinalData().repetition_start;
282 
283  if (l_rep_start != r_rep_start) {
284  // OR node では repetition_start が小さい順、AND node では repetition_start が大きい順に並べたい
285  if (!or_node_ ^ (l_rep_start < r_rep_start)) {
286  return Ordering::kLess;
287  } else {
288  return Ordering::kGreater;
289  }
290  }
291  }
292 
293  if (lhs.Amount() < rhs.Amount()) {
294  return Ordering::kLess;
295  } else if (lhs.Amount() > rhs.Amount()) {
296  return Ordering::kGreater;
297  }
298 
299  return Ordering::kEquivalent;
300  }

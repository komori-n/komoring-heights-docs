144  const auto normal_bytes = static_cast<std::uint64_t>(static_cast<double>(new_bytes) * kNormalRepetitionRatio);
145  const auto rep_bytes = new_bytes - normal_bytes;

385  double GetNormalTableHashRate() const noexcept {
386  // entries_ の最初と最後はエントリ数が若干少ないので、真ん中から kHashfullCalcEntries 個のエントリを調べる
387  const std::size_t begin_idx = Cluster::kSize;
388  const std::size_t count_range_size = cluster_head_num_ - begin_idx;
389 
390  std::size_t used_count = 0;
391  std::size_t idx = begin_idx;
392  for (std::size_t i = 0; i < kHashfullCalcEntries; ++i) {
393  if (!entries_[idx].IsNull()) {
394  used_count++;
395  }
396 
397  // 連続領域をカウントすると偏りが出やすくなってしまうので、大きめの値を足す。
398  idx += 334;
399  if (idx > cluster_head_num_) {
400  idx -= count_range_size;
401  }
402  }

415  auto cluster_end = [&](std::ptrdiff_t start_idx) {
416  return static_cast<std::ptrdiff_t>(std::min(entries_.size(), start_idx + Cluster::kSize));
417  };

420  const auto end_idx = cluster_end(start_idx);
421  const auto used_count = std::count_if(entries_.begin() + start_idx, entries_.begin() + end_idx,
422  [](const Entry& entry) { return !entry.IsNull(); });
423  return static_cast<std::size_t>(used_count);
424  };

426  std::ptrdiff_t start_idx = 0;
427  auto used_in_range = count_used(0);
428  do {

205  constexpr Ordering operator()(const SearchResult& lhs, const SearchResult& rhs) const noexcept {
206  if (lhs.Phi(or_node_) != rhs.Phi(or_node_)) {
207  if (lhs.Phi(or_node_) < rhs.Phi(or_node_)) {
208  return Ordering::kLess;
209  } else {
210  return Ordering::kGreater;
211  }
212  } else if (lhs.Delta(or_node_) != rhs.Delta(or_node_)) {
213  if (lhs.Delta(or_node_) < rhs.Delta(or_node_)) {
214  return Ordering::kLess;
215  } else {
216  return Ordering::kGreater;
217  }
218  }
219 
220  if (lhs.Dn() == 0 /* && rhs.Dn() == 0 */) {
221  const auto l_rep_start = lhs.GetFinalData().repetition_start;
222  const auto r_rep_start = rhs.GetFinalData().repetition_start;
223 
224  if (l_rep_start != r_rep_start) {
225  // OR node では repetition_start が小さい順、AND node では repetition_start が大きい順に並べたい
226  if (!or_node_ ^ (l_rep_start < r_rep_start)) {
227  return Ordering::kLess;
228  } else {
229  return Ordering::kGreater;
230  }
231  }
